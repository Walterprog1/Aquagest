import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')

Deno.serve(async (req) => {
  const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY)
  let payload = {}

  const logResultado = async (msg, exito = false) => {
    await supabase.from('logs_webhook').insert([{
      payload: { ...payload, url: req.url },
      resultado: msg,
      exito
    }])
    return new Response(msg, { status: 200 })
  }

  try {
    const url = new URL(req.url)
    let texto = url.searchParams.get('texto')
    let monto = url.searchParams.get('monto')

    if (!texto) {
      try {
        payload = await req.json()
        texto = payload.texto
        monto = payload.monto
      } catch (e) {}
    }

    if (!texto) return logResultado('No se recibio texto de notificacion', false)

    const textoMin = texto.toLowerCase()

    let montoLimpio = 0
    const regexMonto = /[\$]\s?(\d+[\d.,]*)|(\b\d{1,3}(\.\d{3})+(,\d+)?\b)|(\b\d{3,}(,\d+)?\b)/
    const match = String(monto || texto).match(regexMonto)

    if (match) {
      let textoMonto = match[0].replace(/[\$]/g, '').trim()
      if (textoMonto.includes('.') && textoMonto.includes(',')) {
        montoLimpio = parseFloat(textoMonto.replace(/\./g, '').replace(',', '.'))
      } else if (textoMonto.includes(',')) {
        montoLimpio = parseFloat(textoMonto.replace(',', '.'))
      } else if (textoMonto.includes('.') && textoMonto.split('.').pop().length !== 2) {
        montoLimpio = parseFloat(textoMonto.replace(/\./g, ''))
      } else {
        montoLimpio = parseFloat(textoMonto)
      }
    }

    if (!montoLimpio) return logResultado(`Monto no detectado en: "${texto}"`, false)

    const { data: clientes } = await supabase.from('clientes').select('id, nombre, alias_transferencia')

    const clienteEncontrado = clientes.find(c => {
      const opciones = [c.nombre, ...(c.alias_transferencia ? c.alias_transferencia.split(',') : [])]
      return opciones.some(opcion => {
        const palabras = opcion.toLowerCase().trim().split(/\s+/).filter(p => p.length > 2)
        if (palabras.length === 0) return false
        return palabras.every(palabra => textoMin.includes(palabra))
      })
    })

    if (!clienteEncontrado) return logResultado(`Cliente no identificado: ${texto}`, false)

    const { data: pedidos, error } = await supabase
      .from('pedidos')
      .select('id, total')
      .eq('cliente_id', clienteEncontrado.id)
      .eq('pago_estado', 'pendiente')
      .eq('total', montoLimpio)

    if (error) throw error

    if (pedidos.length > 0) {
      await supabase.from('pedidos').update({ 
        pago_estado: 'pagado',
        notas: `Auto-Match: ${texto}`
      }).eq('id', pedidos[0].id)

      return logResultado(`PAGO OK: ${clienteEncontrado.nombre} ($${montoLimpio})`, true)
    }

    return logResultado(`No hay pedido por $${montoLimpio} para ${clienteEncontrado.nombre}`, false)

  } catch (err) {
    return logResultado(`Error: ${err.message}`, false)
  }
})
